CREATE TABLE CLIENT (
    ID       INTEGER primary key,
    CL_NAME  VARCHAR(32) NOT NULL
);
CREATE TABLE COUNTRY (
    ID           INTEGER primary key,
    COUNTRYNAME  VARCHAR(24) NOT NULL,
    CONTINENT    VARCHAR(30) NOT NULL
);
CREATE TABLE EXCURSION (
    ID           INTEGER primary key,
    NAME         VARCHAR(25) NOT NULL,
    TOURFIRM_ID  INTEGER NOT NULL,
    EXC_COST     INTEGER
);
CREATE TABLE FEED (
    ID         INTEGER primary key,
    FEED_TYPE  VARCHAR(25) NOT NULL
);
CREATE TABLE HOTEL (
    ID          INTEGER primary key,
    HOTELNAME   VARCHAR(22) NOT NULL,
    COUNTRY_ID  INTEGER NOT NULL,
    STARS       INTEGER NOT NULL
);
CREATE TABLE ROOM (
    ID           INTEGER primary key,
    HOTEL_ID     INTEGER NOT NULL,
    ROOM_NUMBER  INTEGER NOT NULL
);
CREATE TABLE TOUR (
    ID           INTEGER primary key,
    HOTEL_ID     INTEGER NOT NULL,
    TOURFIRM_ID  INTEGER NOT NULL,
    FEED_ID    INTEGER,
    TOUR_COST    INTEGER,
    DURATION     INTEGER
);
CREATE TABLE TOURFIRM (
    ID          INTEGER primary key,
    TFIRM_NAME  VARCHAR(25) NOT NULL
);
CREATE TABLE ORDER_TOUR (
    ID 		 INTEGER primary key,
    CL_ID	 INTEGER NOT NULL,
    TOUR_ID 	 INTEGER NOT NULL,
    DATE_IN 	 DATE,
    ORDER_COST  INTEGER
);
CREATE TABLE TOUR_EXCURSION (
    TOUR_ID	INTEGER NOT NULL,
    EXC_ID	INTEGER not null
);
