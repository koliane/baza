create table airflight (
id integer primary key,
country_id integer not null,
cost integer);
alter table order_tour add airflight_id int not null;
alter table order_tour add constraint ordertour_to_airflight foreign key (airflight_id) references airflight (ID);

create table transfer (
id integer primary key,
hotel_id integer,
type_TR varchar(25)
);
alter table order_tour add transfer_id int not null;
alter table order_tour add constraint ordertour_to_transfer foreign key (transfer_id) references transfer (ID);


create table order_tour_clients (
id integer primary key,
client_id integer not null,
order_tour_id integer not null
);

alter table order_tour add order_tour_clients_id int not null;
alter table order_tour add constraint ordertour_to_clients foreign key (order_tour_clients_id) references order_tour_clients (ID);

alter table order_tour_clients add constraint ordertourcl_to_client foreign key (client_id) references client(ID);
alter table order_tour_clients add constraint ordertourcl_to_ordertour foreign key (order_tour_id) references order_tour(ID);
